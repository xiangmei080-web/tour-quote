<script>
let lastSum = 0;
let lastDate = "未載日期";
const THEME_KEY = "p_quote_theme_v1";
const MODE_KEY = "p_quote_mode_v1";
const holidayList = [
    { start: "04/02", end: "04/06", name: "清明連假" },
    { start: "05/01", end: "05/01", name: "勞動節" },
    { start: "06/19", end: "06/21", name: "端午連假" },
    { start: "09/25", end: "09/28", name: "中秋連假" },
    { start: "10/09", end: "10/11", name: "國慶連假" }
];
const weekDayMap = ["日", "一", "二", "三", "四", "五", "六"];
function applySavedAppearance() {
    const savedTheme = localStorage.getItem(THEME_KEY) || "pink";
    const savedMode = localStorage.getItem(MODE_KEY) || "light";
    setTheme(savedTheme, false);
    setDarkMode(savedMode === "dark", false);
}
function setTheme(themeName, save = true) {
    document.body.classList.remove("theme-lavender", "theme-mint");
    if (themeName === "lavender") document.body.classList.add("theme-lavender");
    if (themeName === "mint") document.body.classList.add("theme-mint");
    const picker = document.getElementById("themePicker");
    if (picker) picker.value = themeName;
    if (save) localStorage.setItem(THEME_KEY, themeName);
}
function setDarkMode(isDark, save = true) {
    document.body.classList.toggle("dark-mode", isDark);
    const btn = document.getElementById("modeToggle");
    if (btn) btn.innerText = isDark ? "☀️ 淺色" : "🌙 深色";
    if (save) localStorage.setItem(MODE_KEY, isDark ? "dark" : "light");
}
function toggleDarkMode() {
    setDarkMode(!document.body.classList.contains("dark-mode"));
}
function resetAppearance() {
    setTheme("pink");
    setDarkMode(false);
}
function safeEval(expr) {
    const cleaned = String(expr || "").replace(/\s+/g, "");
    if (!cleaned || !/^[\d+\-*/().]+$/.test(cleaned)) return null;
    try {
        const v = new Function("return " + cleaned)();
        return (typeof v === "number" && isFinite(v)) ? v : null;
    } catch (e) {
        return null;
    }
}

function checkDateType(dateStr) {
    if (!dateStr) return { text: "", dObj: null };
    const p = dateStr.trim().split('/');
    if (p.length !== 2) return { text: "", dObj: null };
    const dObj = new Date(2026, parseInt(p[0]) - 1, parseInt(p[1]));
    const m = (dObj.getMonth() + 1).toString().padStart(2, '0');
    const d = dObj.getDate().toString().padStart(2, '0');
    const checkStr = `${m}/${d}`;
    const dayName = weekDayMap[dObj.getDay()];
    let holidayName = "";
    for (let h of holidayList) {
        if (checkStr >= h.start && checkStr <= h.end) {
            holidayName = `🚩 ${h.name}`;
            break;
        }
    }
    let typeText = holidayName ? `${holidayName} (週${dayName})` : `${([0, 5, 6].includes(dObj.getDay())) ? "☀️ 假日" : "☕ 平日"} (週${dayName})`;
    return { text: typeText, dObj: dObj };
}

function doSync() {
    const input = document.getElementById("input").value;
    const name = document.getElementById("custName").value;
    const resDiv = document.getElementById("result");
    if(!input.trim()) { resDiv.innerText = ""; lastSum = 0; return; }

    const lines = input.split("\n");
    let out = name ? `👤 致：${name} 貴賓\n` : "";
    out += "📋 澎湖旅遊行程報價明細\n──────────────\n\n";
    
    let totalSum = 0; 
    let groupSum = 0; 

    lines.forEach(line => {
        let text = line.trim();
        if(!text) return; 

        let dateRange = text.match(/^(\d{1,2}\/\d{1,2})\s*[-~～]\s*(\d{1,2}\/\d{1,2})$/);
        let singleDate = text.match(/^(\d{1,2}\/\d{1,2})$/);

        if((dateRange || singleDate) && !text.includes("*")) {
            lastDate = text;
            if(dateRange) {
                const s = checkDateType(dateRange[1]);
                const e = checkDateType(dateRange[2]);
                let duration = (s.dObj && e.dObj) ? ` 【${Math.round((e.dObj - s.dObj) / 86400000) + 1}天${Math.round((e.dObj - s.dObj) / 86400000)}夜】` : "";
                out += `📅 日期：${text}${duration}\n   去程：${s.text}\n   回程：${e.text}\n\n`;
            } else {
                out += `📅 日期：${text}\n   類型：${checkDateType(singleDate[1]).text}\n\n`;
            }
            return;
        }

        let ageRange = text.match(/^(\d{1,2})\s*-\s*(\d{1,2})\s*歲?$/);
        if (ageRange) {
            out += `👶 兒童年齡：${ageRange[1]}-${ageRange[2]}歲\n`;
            return;
        }

        // 純數學算式（例如 5000+2000*2）直接計算，不當作「單價*人數」
        if (/^[\d+\-*/().\s]+$/.test(text)) {
            let v = safeEval(text);
            if (v !== null) {
                out += `● 費用：$${v.toLocaleString()}\n`;
                groupSum += v;
                totalSum += v;
                return;
            }
        }

        if(text.includes("*")) {
            let parts = text.split("*");
            let labelWithMath = parts[0].trim();
            let qty = parseFloat(parts[1]) || 1;
            let mathMatch = labelWithMath.match(/([\d\+\-\(\)\. ]+)$/);
            if(mathMatch) {
                try {
                    let formula = mathMatch[0].trim();
                    let unitPrice = new Function("return " + formula)();
                    let label = labelWithMath.replace(formula, "").trim();
                    if (!label) return;
                    let total = unitPrice * qty;
                    out += `● ${label}：$${unitPrice.toLocaleString()} × ${qty}位 = $${total.toLocaleString()}\n`;
                    groupSum += total; 
                    totalSum += total;
                    return;
                } catch(e) {}
            }
        }

        if((text.includes("優惠") || text.includes("折扣") || (text.startsWith("-") && text.match(/\d/))) && !text.includes("*")) {
            let valMatch = text.match(/-?\d+/);
            if(valMatch) {
                let rawVal = parseInt(valMatch[0]);
                // 關鍵字項目視為扣款，避免「折扣-192」被解析成 +192
                let val = (text.includes("優惠") || text.includes("折扣")) ? -Math.abs(rawVal) : rawVal;
                let label = text.replace(valMatch[0], "").trim().replace(/[-:：\s]+$/, "");
                if (!label) label = "折扣";
                out += `✨ ${label}：$${val.toLocaleString()}\n`;
                groupSum += val;
                totalSum += val;
                return;
            }
        }

        let labelExpr = text.match(/^([^\d]+)\s*([\d+\-*/(). ]+)$/);
        if (labelExpr) {
            let label = labelExpr[1].trim().replace(/[:：]$/, "");
            let v = safeEval(labelExpr[2]);
            if (v !== null) {
                out += `● ${label}：$${v.toLocaleString()}\n`;
                groupSum += v;
                totalSum += v;
                return;
            }
        }

        if(text === "小計") {
            out += `💰 【本單小計】：NT$ ${groupSum.toLocaleString()}\n`;
            out += `🛡️ 費用包含 旅責險 意外250萬 醫療20萬\n`;
            out += `📜 規範：訂單成立比照國內定型化契約進行。\n──────────────\n\n`;
            groupSum = 0; 
            return;
        }

        out += text + "\n";
    });
    
    lastSum = totalSum;
    resDiv.innerText = out;
}

function exportToFile() {
    const text = document.getElementById("result").innerText;
    const name = document.getElementById("custName").value || "未命名";
    if(!text) { alert("尚未產生報價內容"); return; }
    const blob = new Blob([text], { type: "text/plain" });
    const anchor = document.createElement("a");
    anchor.download = `${name}_報價單.txt`;
    anchor.href = window.URL.createObjectURL(blob);
    anchor.click();
}

function saveArchive() {
    const name = document.getElementById("custName").value || "未命名";
    const rawInput = document.getElementById("input").value;
    if(!rawInput.trim()) return;
    doSync();
    const archive = JSON.parse(localStorage.getItem("p_quotes_v2") || "[]");
    archive.unshift({ id: Date.now(), name, date: lastDate, sum: lastSum, rawInput });
    localStorage.setItem("p_quotes_v2", JSON.stringify(archive.slice(0, 50)));
    renderArchive();
    alert("💾 已成功存檔");
}

function renderArchive() {
    const archive = JSON.parse(localStorage.getItem("p_quotes_v2") || "[]");
    const listDiv = document.getElementById("archiveList");
    listDiv.innerHTML = archive.map(q => `
        <div class="archive-item" onclick="loadQuote(${q.id})">
            <button class="del-btn" onclick="deleteQuote(event, ${q.id})">×</button>
            <span class="arch-name">${q.name}</span>
            <span class="arch-sum">${q.date} | $${(q.sum || 0).toLocaleString()}</span>
        </div>
    `).join("") || '<div style="text-align:center;color:#999;padding:20px;">暫無存檔</div>';
}

function loadQuote(id) {
    const archive = JSON.parse(localStorage.getItem("p_quotes_v2") || "[]");
    const q = archive.find(i => i.id === id);
    if(q) {
        document.getElementById("custName").value = q.name;
        document.getElementById("input").value = q.rawInput;
        doSync();
    }
}

function deleteQuote(e, id) {
    e.stopPropagation();
    if(!confirm("確定刪除？")) return;
    let archive = JSON.parse(localStorage.getItem("p_quotes_v2") || "[]");
    localStorage.setItem("p_quotes_v2", JSON.stringify(archive.filter(i => i.id !== id)));
    renderArchive();
}

function clearAll() {
    if(confirm("確定清空？")) {
        document.getElementById("custName").value = "";
        document.getElementById("input").value = "";
        document.getElementById("result").innerText = "";
    }
}

function copy() {
    const text = document.getElementById("result").innerText;
    if(!text) return;
    navigator.clipboard.writeText(text).then(() => alert("✅ 報價已複製"));
}

    applySavedAppearance();
    renderArchive();
